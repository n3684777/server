amatsu,123,160,4	script	極速快感主持人	985,{

	if($@handfast == 1)
	{
		warp "quiz_02",363,345;
		end;
	}
	if ( getgmlevel() != 0 )goto L1;

	else

	{

		mes "[極速快感主持人]";
		mes "現在不是開放時間喔";
		mes "每日的6點到晚上12點";
		mes "每個2小時都有一場";
		mes "感謝您的參予^^";
		close;
	}
L1:
	mes "[極速快感主持人]";
	mes "歡迎進入GM控制選單";
	mes "在這裡你可以隨時啟動遊戲";	
	mes "最好不要在遊戲原本固定的時間開啟";	
	mes "以免發生錯誤!";	
	next;
        menu "離開",-,"開啟遊戲.",startgame;
	close;

startgame:
	if($@handok != 0) close;
	mes "[極速快感主持人]";
	mes "啟動遊戲!";
	close2;
	set $@handfast,1;
	announce "[極速快感] : 極速快感遊戲開始-請開始入場",bc_all;
	sleep2 2000;
	announce "[極速快感] : 極速快感遊戲開始-在三分鐘關閉入口",bc_all;
	sleep2 120000;
	announce "[極速快感] : 極速快感遊戲開始-在一分鐘關閉入口",bc_all;
	sleep2 60000;
	set $@handfast,0;
	announce "[極速快感] : 極速快感遊戲已經開始-關閉入場!",bc_all;
	sleep2 3000;
	donpcevent "handgame::OnStart";
	end;
OnInit:
	set $fcanpass,2;
	waitingroom "極速快感",0;
	end;

}
